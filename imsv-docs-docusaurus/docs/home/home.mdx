<h2>Endpoints</h2>

```mdx-code-block
import DocCardList from '@theme/DocCardList';
import sidebars from '../../sidebars.js';

export const endpointSubcategories = (() => {
  // Find the "Endpoints" category
  const endpointCategory = sidebars.primarySidebar.find(
    item => item.type === 'category' && item.label === 'Endpoints'
  );
  if (!endpointCategory) return [];

  const firstLevelCategories = (Array.isArray(endpointCategory.items)
    ? endpointCategory.items
    : endpointCategory.items.default
  ).filter(item => item.type === 'category');

  return firstLevelCategories.map(cat => ({
    type: 'link',
    label: cat.label,
    href: cat.link?.type === 'doc' ? `/api-reference/${cat.link.id}` : '#',
  }));
})();

<DocCardList items={endpointSubcategories} />
```

<h1>Webhook Topics</h1>


```mdx-code-block

export const webhookSubcategories = (() => {
  // Find the "Webhook Topics" category
  const webhookCategory = sidebars.primarySidebar.find(
    item => item.type === 'category' && item.label === 'Webhook Topics'
  );

  if (!webhookCategory) return [];

  // Get first-level children that are categories
  const firstLevelCategories = (Array.isArray(webhookCategory.items)
    ? webhookCategory.items
    : webhookCategory.items.default
  ).filter(item => item.type === 'category');

  // Map them into DocCardList items
  return firstLevelCategories.map(cat => ({
    type: 'link',
    label: cat.label,
    href: cat.link && cat.link.type === 'doc'
      ? `/api-reference/${cat.link.id}`
      : '#', // fallback if no link
  }));
})();

<DocCardList items={webhookSubcategories} />
```
